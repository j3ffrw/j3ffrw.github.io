Export email address list from cpanel

  1. install firebug on firefox. install also if firequery if Cpanel doesnot use jquery.

  2. login to cpanel and go to Mail accounts, view maximum emails possible.

  3. paste the following code to firebug command line

    var rows = $("#table_email_accts tr.dt_info_row"),
returnString = "";
rows.each(function(i, item){
var tdList = [];
$(item).find('td:eq(0)').each(function(s, subItem){
tdList.push($(subItem).text());
});
returnString += tdList + "\n";
});
console.log(returnString);
